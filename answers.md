### 12. –°–∫–æ–ª—å–∫–æ –≤—Ä–µ–º–µ–Ω–∏ –≤ –º–∏–Ω—É—Ç–∞—Ö –∑–∞–π–º–µ—Ç —É –≤–∞—Å –Ω–∞–ø–∏—Å–∞–Ω–∏–µ –ø—Ä–æ—Ü–µ–¥—É—Ä—ã –æ–±—Ä–∞—â–µ–Ω–∏—è –æ–¥–Ω–æ—Å–≤—è–∑–Ω–æ–≥–æ —Å–ø–∏—Å–∫–∞?

```go
type item struct {
	data int
	next *item
}

func (head *item) appendElement(element int) {
	p := head
	for p.next != nil {
		p = p.next
	}
	p.next = newItem(element)
}

func newItem(data int) *item {
	return &item{
		data: data,
	}
}

func (head *item) printList() {
	for p := head; p != nil; p = p.next {
		fmt.Printf("%d ", p.data)
	}
	fmt.Println()
}

func (head *item) Reverse() *item {
	var prev *item = nil
	current := head
	var next *item = nil
	for current != nil {
		next = current.next
		current.next = prev
		prev = current
		current = next
	}
	return prev
}
```
